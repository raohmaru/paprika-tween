    Example

    import { Spice } from 'paprika-tween';
import { Elastic } from 'paprika-tween/easing';
const spice = new Spice({
    duration: 120,
    easing: Elastic.InOut,
    from: { rotation: 10 },
    to: { rotation: 520 },
    render: ({ rotation }) => { ... }
});
spice.start();
(function loop(timestamp) {
    spice.frame(timestamp);
    requestAnimationFrame(loop);
})()
